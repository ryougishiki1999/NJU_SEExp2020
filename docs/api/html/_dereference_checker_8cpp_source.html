29  const clang::Expr *expr, const clang::ento::ProgramState *state,
30  const clang::LocationContext *lctx, bool loaded_from)
31 {
32  //wipe out parentheses and l value casts
33  expr = expr->IgnoreParenLValueCasts();
34  switch (expr->getStmtClass())
35  {
36  //when the class of error statement is a declation reference
37  //expression, add the source infomation of the error variable
38  case clang::Stmt::DeclRefExprClass: {
39  const clang::DeclRefExpr *decl_ref_expr =
40  clang::cast<clang::DeclRefExpr>(expr);
41  if (const clang::VarDecl *var_decl =

43  {
44  os << " (" << (loaded_from ? "loaded from" : "from")

48  break;
49  }

51  // the source infomation of the error field
52  case clang::Stmt::MemberExprClass: {

58  break;
59  }
60  default:
61  break;
62  }

83 // report direct null dereferenced
84 void DereferenceChecker::report_direct_dereference(clang::ento::ProgramStateRef state, const clang::Stmt *stmt,
85  clang::ento::CheckerContext &ctx) const
86 {
87  // init direct
88  if (!direct)
89  direct.reset(new clang::ento::BugType(this, "Null-Pointer-Dereferenced", clang::ento::categories::MemoryError));
90  // generate an error node
91  clang::ento::ExplodedNode *node = ctx.generateErrorNode(state);
92  if (!node)
93  return;
94  //add some infomation for report
95  clang::SmallString<100> buf;
96  llvm::raw_svector_ostream os(buf);
97  clang::SmallVector<clang::SourceRange, 2> ranges;
98  switch (stmt->getStmtClass())
99  {
100  // when the class of the error statement is array subscript expression
101  // add the array access information for error description
102  case clang::Stmt::ArraySubscriptExprClass: {
103  os << "Array access";
104  const clang::ArraySubscriptExpr *array_sub_expr = clang::cast<clang::ArraySubscriptExpr>(stmt);
105  add_deref_source(os, ranges, array_sub_expr->getBase()->IgnoreParenCasts(), state.get(),
106  node->getLocationContext());
107  os << " results in a null pointer dereference";
108  break;
109  }
110  // when the class of the error statement is unary operator expression
111  // add information for error description
112  case clang::Stmt::UnaryOperatorClass: {
113  os << "Dereference of null pointer";
114  const clang::UnaryOperator *unary_operator = clang::cast<clang::UnaryOperator>(stmt);
115  add_deref_source(os, ranges, unary_operator->getSubExpr()->IgnoreParens(), state.get(),
116  node->getLocationContext(), true);
117  break;
118  }
119  // when the class of the error statement is member expression
120  // add the field access information for error description
121  case clang::Stmt::MemberExprClass: {
122  const clang::MemberExpr *member_expr = clang::cast<clang::MemberExpr>(stmt);
123  if (member_expr->isArrow() || is_decl_ref_expr_to_reference(member_expr->getBase()))
124  {
125  os << "Access to field '" << member_expr->getMemberNameInfo()
126  << "' results in a dereference of a null pointer";
127  add_deref_source(os, ranges, member_expr->getBase()->IgnoreParenCasts(), state.get(),
128  node->getLocationContext(), true);
129  }
130  break;
131  }
132  default:
133  break;
134  }
135  // generate report
136  auto report = std::make_unique<clang::ento::PathSensitiveBugReport>(
137  *direct, buf.empty() ? "Dereference of null pointer" : clang::StringRef(buf), node);
138  clang::ento::bugreporter::trackExpressionValue(node, clang::ento::bugreporter::getDerefExpr(stmt), *report);
139  //add ranges for error report
140  for (clang::SmallVectorImpl<clang::SourceRange>::iterator
141  i = ranges.begin(), e = ranges.end(); i != e; i++)
142  {
143  report->addRange(*i);
144  }
145  ctx.emitReport(std::move(report));
146 }

172  // an implicit null dereference. if report it, it results in a high

175  }

181 void DereferenceChecker::checkBind(clang::ento::SVal l, clang::ento::SVal v, const clang::Stmt *stmt,
182  clang::ento::CheckerContext &ctx) const
183 {
184  // if the value is undefined
185  if (v.isUndef())
186  return;
187  const clang::ento::MemRegion *region = l.getAsRegion();
188  const clang::ento::TypedValueRegion *typed_value_region =
189  clang::dyn_cast_or_null<clang::ento::TypedValueRegion>(region);
190  if (!typed_value_region)
191  return;
192  // check if l is a reference type
193  if (!typed_value_region->getValueType()->isReferenceType())
194  return;
195  // get program state
196  clang::ento::ProgramStateRef state = ctx.getState();
197  clang::ento::ProgramStateRef not_null_state, null_state;
198  // judge the value of location can be null or nonnull or both
199  std::tie(not_null_state, null_state) = state->assume(v.castAs<clang::ento::DefinedOrUnknownSVal>());
200  if (null_state)
201  {
202  // explicit NULL case
203  if (!not_null_state)
204  {
205  const clang::Expr *expr = get_dereference_expr(stmt, true);
206  report_direct_dereference(null_state, expr, ctx);
207  return;
208  }
209  // the value of v is unknown(either null or nonnull),there maybe
210  // an implicit null dereference. if report it, it results in a high
211  // false alarm rate and blocks detection of other errors
212  }
213  //report the error when the l is used.like this:
214  //int &b=*a;
215  //if(a!=NULL) return;
216  //int c=b; error!
217  //to achieve this effect, do not transite the program state.
218  ctx.addTransition(state, this);
219 }
