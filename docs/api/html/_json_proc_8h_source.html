41 static std::string make_indent(size_t indent)
42 {
43  static bool is_init = false;
44  const int bufsize = 10;
45  static std::string buf[bufsize];
46 
47  if (!is_init)
48  {
49  for (int i = 0; i < bufsize; i++)
50  {
51  for (int j = 0; j < i; j++)
52  {
53  buf[i] += '\t';
54  }
55  }
56  is_init = true;
57  }
58  if (indent < bufsize)
59  return buf[indent];
60  else
61  {
62  std::string str;
63  while (indent-- > 0)
64  {
65  str += '\t';
66  }
67  return str;
68  }

172  /*
173  \sa Value::append
174  */
175  void append(const std::string &key, Value &v) override
176  {
177  attributes.insert_or_assign(key, &v);
178  attributes_vec.push_back(std::make_pair(key, &v));
179  }
180 
183  std::string to_string(size_t indent = 0, bool cont_line = true) const override
184  {
185  std::string str;
186  if (cont_line)
187  str += "\n";
188  str += make_indent(indent) + "{\n";
189  for (int i = 0; i < attributes_vec.size(); i++)
190  {
191  const std::pair<std::string, Value *> &p = attributes_vec[i];
192  const std::string name = p.first;
193  const Value *v = p.second;
194 
195  str += make_indent(indent + 1) + "\"" + name + "\"" + " : " + v->to_string(indent + 2, true);
196 
197  if (i != attributes_vec.size() - 1)
198  str += ",";
199  str += "\n";
200  }
201  str += make_indent(indent) + "}";
202  return str;
203  }
204 
207  void release() override
208  {
209  for (std::pair<std::string, Value *> p : attributes_vec)
210  {
211  p.second->release();
212  delete p.second;
213  }
