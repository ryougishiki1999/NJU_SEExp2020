30 void MemoryChecker::checkDeadSymbols(clang::ento::SymbolReaper &sym_reaper, clang::ento::CheckerContext &ctx) const
31 {
32  // check memory leak
33  clang::ento::ProgramStateRef state = ctx.getState();
34  MappedMemoryStateTy ms = state->get<MappedMemoryState>();
35  MappedMemoryStateTy::Factory &ms_f = state->get_context<MappedMemoryState>();
36  std::vector<clang::ento::SymbolRef> leak_memory;
37  MappedMemoryStateTy new_ms = ms;
38  for (auto i : new_ms)
39  {
40  if (sym_reaper.isDead(i.first))
41  {
42  if (i.second.state == MemoryState::Allocated)
43  leak_memory.push_back(i.first);
44  new_ms = ms_f.remove(new_ms, i.first);
45  }
46  }
47  if (new_ms == ms)
48  return;
49  clang::ento::ExplodedNode *node = ctx.getPredecessor();
50  if (!leak_memory.empty())
51  {
52  static clang::ento::CheckerProgramPointTag tag("MemoryChecker", "MemoryLeak");
53  node = ctx.generateNonFatalErrorNode(ctx.getState(), &tag);
54  if (node)
55  {
56  for (auto &i : leak_memory)
57  {
58  report_memory_leak(ctx, i, node);
59  }
60  }
61  }
62  ctx.addTransition(state->set<MappedMemoryState>(new_ms), node);
63 }
64 void MemoryChecker::checkPreCall(const clang::ento::CallEvent &call, clang::ento::CheckerContext &ctx) const
65 {
66  // skip the destructor, since we check delete in postcall
67  if (llvm::dyn_cast<clang::ento::CXXDestructorCall>(&call))
68  {
69  return;
70  }
71  // check whether call is free call to check double free in post call
72  if (const clang::ento::AnyFunctionCall *c = llvm::dyn_cast<clang::ento::AnyFunctionCall>(&call))
73  {
74  const clang::FunctionDecl *decl = c->getDecl();
75  if (!decl)
76  return;
77  if (deallocate_function_map.lookup(call))
78  {
79  return;
80  }
81  if (is_new_delete(decl))
82  {
83  return;
84  }
85  }
86  // check callee
87  if (const clang::ento::CXXInstanceCall *c = llvm::dyn_cast<clang::ento::CXXInstanceCall>(&call))
88  {
89  clang::ento::SymbolRef memory = c->getCXXThisVal().getAsSymbol();
90  if (!memory || check_use_after_free(ctx, memory, c->getCXXThisExpr()))
91  return;
92  }
93  // check args
94  for (auto i = 0; i < call.getNumArgs(); ++i)
95  {
96  clang::ento::SVal arg = call.getArgSVal(i);
97  if (arg.getAs<clang::ento::Loc>())
98  {
99  clang::ento::SymbolRef memory = arg.getAsSymbol();
100  if (memory && check_use_after_free(ctx, memory, call.getArgExpr(i)))
101  return;
102  }
103  }

105 void MemoryChecker::checkPostCall(const clang::ento::CallEvent &call, clang::ento::CheckerContext &ctx) const
106 {
107  if (ctx.wasInlined)
108  return;
109  const clang::CallExpr *expr = llvm::dyn_cast_or_null<clang::CallExpr>(call.getOriginExpr());
110  if (!expr)
111  return;
112  const clang::FunctionDecl *decl = ctx.getCalleeDecl(expr);
113  if (!decl)
114  return;
115  if (const MemoryChecker::CheckFunctionType *func = allocate_function_map.lookup(call))
116  {
117  (*func)(this, ctx, expr, ctx.getState());
118  return;
119  }
120  if (const MemoryChecker::CheckFunctionType *func = deallocate_function_map.lookup(call))
121  {
122  (*func)(this, ctx, expr, ctx.getState());
123  return;
124  }
125  // for new/delete call
126  // e.g. void *p = operator new(sizeof(int));
127  // operator delete(p);
128  if (is_new_delete(llvm::cast<clang::FunctionDecl>(call.getDecl())))
129  {
130  check_new_delete(ctx, expr, ctx.getState());
131  return;
132  }

212 clang::ento::ProgramStateRef MemoryChecker::handle_allocation(clang::ento::CheckerContext &ctx,
213  const clang::CallExpr *call_expr,
214  clang::ento::ProgramStateRef state,
215  MemoryState::Function alloc_func) const
216 {
217  // track the allocation.
218  // invalid state
219  if (!state)
220  return nullptr;
221  // invalid return value
222  if (!clang::ento::Loc::isLocType(call_expr->getType()))
223  return nullptr;
224  // bind effect to the allocation stmt
225  // bind heap region symbolic value.
226  clang::ento::SValBuilder &sb = ctx.getSValBuilder();
227  auto loc_ctx = ctx.getPredecessor()->getLocationContext();
228  clang::ento::DefinedSVal memory =
229  sb.getConjuredHeapSymbolVal(call_expr, loc_ctx, ctx.blockCount()).castAs<clang::ento::DefinedSVal>();
230  state = state->BindExpr(call_expr, ctx.getLocationContext(), memory);
231  return register_allocation_memory_state(ctx, call_expr, state, alloc_func, ctx.getSVal(call_expr));
232 }
233 clang::ento::ProgramStateRef MemoryChecker::handle_deallocation(clang::ento::CheckerContext &ctx,
234  const clang::Expr *call_expr,
235  const clang::Expr *arg_expr,
236  clang::ento::ProgramStateRef state) const
237 {
238  if (!state)
239  return nullptr;
240  clang::ento::SVal pointer = ctx.getSVal(arg_expr);
241  // check region
242  const clang::ento::MemRegion *region = pointer.getAsRegion();
243  if (!region)
244  return nullptr;
245  region = region->StripCasts();
246  const clang::ento::SymbolicRegion *sym_region =
247  llvm::dyn_cast<clang::ento::SymbolicRegion>(region->getBaseRegion());
248  if (!sym_region)
249  return nullptr;
250  clang::ento::SymbolRef memory = sym_region->getSymbol();
251  auto memory_state = state->get<MappedMemoryState>(memory);
252  if (memory_state)
253  {
254  if (memory_state->state == MemoryState::Deallocated)
255  {
256  report_double_free(ctx, call_expr->getSourceRange(), memory);
257  }
258  else
259  {
260  // TODO MAKE ILLEGAL <ALLOCATION,DEALLOCATION> PAIR AS MEMORY LEAK
261  }
262  // update memory state
263  return state->set<MappedMemoryState>(
264  memory, MemoryState(MemoryState::Deallocated, memory_state->alloc_func, memory_state->bind_stmt));
265  }
266  return nullptr;
267 }

340 void MemoryChecker::report_memory_leak(clang::ento::CheckerContext &ctx, clang::ento::SymbolRef memory,
341  clang::ento::ExplodedNode *node) const
342 {
343  if (!memory_leak_bug_type)
344  {
345  memory_leak_bug_type.reset(new clang::ento::BugType(this, "Memory Leak", clang::ento::categories::MemoryError));
346  }
347  // get leak allocation
348  clang::ento::ExplodedNode *alloc_node = node;
349  const clang::ento::MemRegion *region = nullptr;
350  const clang::LocationContext *leak_loc_ctx = node->getLocationContext();
351  for (auto n = node; n;)
352  {
353  clang::ento::ProgramStateRef s = n->getState();
354  if (!s->get<MappedMemoryState>(memory))
355  break;
356  if (!region)
357  {
358  if (const clang::ento::MemRegion *mem_region = ctx.getLocationRegionIfPostStore(n))
359  {
360  if (s->getSVal(mem_region).getAsLocSymbol() == memory)
361  {
362  const clang::ento::VarRegion *var_region =
363  mem_region->getBaseRegion()->getAs<clang::ento::VarRegion>();
364  if (!var_region || (var_region->getStackFrame() == leak_loc_ctx->getStackFrame()))
365  region = mem_region;
366  }
367  }
368  }
369  // backtracing
370  const clang::LocationContext *cur_ctx = n->getLocationContext();
371  if (cur_ctx == leak_loc_ctx || cur_ctx->isParentOf(leak_loc_ctx))
372  alloc_node = n;
373  if (n->pred_empty())
374  n = nullptr;
375  else
376  n = *(n->pred_begin());
377  }
378  // report
379  clang::ento::PathDiagnosticLocation loc;
380  if (const clang::Stmt *s = alloc_node->getStmtForDiagnostics())
381  {
382  loc = clang::ento::PathDiagnosticLocation::createBegin(s, ctx.getSourceManager(),
383  alloc_node->getLocationContext());
384  }
385  std::string msg;
386  llvm::raw_string_ostream os(msg);
387  os << "Potential memory leak";
388  if (region && region->canPrintPretty())
389  {
390  os << " at pointer ";
391  region->printPretty(os);
392  std::unique_ptr<clang::ento::PathSensitiveBugReport> report =
393  std::make_unique<clang::ento::PathSensitiveBugReport>(*memory_leak_bug_type, os.str(), node, loc,
394  alloc_node->getLocationContext()->getDecl());
395  report->markInteresting(memory);
396  ctx.emitReport(std::move(report));
397  }
398 }
